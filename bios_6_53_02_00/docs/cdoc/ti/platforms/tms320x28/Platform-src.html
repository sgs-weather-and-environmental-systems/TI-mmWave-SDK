    36    package ti.platforms.tms320x28;
    37    
    38    /*!
    39     *  ======== Platform ========
    40     *  A generic platform that supports any TMS320x28 device
    41     *
    42     *  The device to be used by this platform is passed as the platform instance
    43     *  name. On the `xdc.tools.configuro` command line, it is done in the
    44     *  following way:
    45     *  @p(code)
    46     *  xs xdc.tools.configuro ... -p "ti.platforms.tms320x28:TMS320C2812"
    47     *  @p
    48     *
    49     *  In package.bld, the platform instance is selected as in:
    50     *  @p(code)
    51     *  Pkg.addExecutable("test", target, "ti.platforms.tms320x28:TMS320C2812");
    52     *  @p
    53     */
    54    metaonly module Platform inherits xdc.platform.IPlatform
    55    {
    56        config xdc.platform.IPlatform.Board BOARD = {
    57            id:             "0",
    58            boardName:      "tms320x28",
    59            boardFamily:    null,
    60            boardRevision:  null
    61        };
    62    
    63        /*!
    64         *  ======== nameFormat ========
    65         *  Encoding of instance creation parameters in the instance's name
    66         *
    67         *  For this platform, the parameters `deviceName`, `includeLinkCmdFile`
    68         *  and `clockRate` can be encoded in the instance name supplied on
    69         *  `xdc.tools.configuro` command line, for example:
    70         *  @p(code)
    71         *      xs xdc.tools.configuro ... -p ti.platforms.tms320x28:TMS320F28035:1:60
    72         *  @p
    73         *  Optional parameters can be omitted:
    74         *  @p(code)
    75         *      xs xdc.tools.configuro ... -p ti.platforms.tms320x28:TMS320F28035
    76         *  @p
    77         */
    78        readonly config string nameFormat
    79            = "$(deviceName):$(includeLinkCmdFile):$(clockRate)";
    80    
    81        /*!
    82         *  ======== genAliases ========
    83         *  Generate symbol mapping for TI Tools 6.0 or newer
    84         */
    85        config Bool genAliases = true;
    86    
    87    instance:
    88    
    89        config xdc.platform.IExeContext.Cpu CPU = {
    90            id:             "0",
    91            clockRate:      0,
    92            catalogName:    "ti.catalog.c2800",
    93            deviceName:     "TMS320F28069",
    94            revision:       null,
    95        };
    96    
    97        /*!
    98         *  ======== deviceName ========
    99         *  The name of an `ICpuDataSheet` module for the device
   100         *
   101         *  This parameter is required, but it does not have to be set explicitly;
   102         *  it can be encoded in the instance's name.
   103         */
   104        config string deviceName;
   105    
   106        /*!
   107         *  ======== clockRate ========
   108         *  The clock rate for this device.
   109         */
   110        config Double clockRate = 0;
   111    
   112        override config string codeMemory = null;
   113    
   114        override config string dataMemory = null;
   115    
   116        override config string stackMemory = null;
   117    
   118        /*!
   119         *  ======== includeLinkCmdFile ========
   120         *  The flag that specifies if the platform should include a linker command
   121         *  file.
   122         *
   123         *  By default, a user is responsible for adding a linker command file to
   124         *  the project, or to the linker command line. However, if this flag is
   125         *  set, this platform will include a default linker command file for the
   126         *  selected device.
   127         */
   128        config Bool includeLinkCmdFile = false;
   129    };
