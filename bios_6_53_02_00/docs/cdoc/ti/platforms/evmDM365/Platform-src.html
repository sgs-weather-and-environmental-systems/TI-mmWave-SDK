    37    /*!
    38     *  ======== Platform ========
    39     *  Platform support for the evmDM365
    40     *
    41     *  This module implements xdc.platform.IPlatform and defines configuration
    42     *  parameters that correspond to this platform's Cpu's, Board's, etc.
    43     *
    44     *  The configuration parameters are initialized in this package's
    45     *  configuration script (package.cfg) and "bound" to the XDC object
    46     *  model.  Once they are part of the model, these parameters are
    47     *  queried by a program's configuration script.
    48     *
    49     *  This particular platform has a single Cpu, and therefore, only
    50     *  declares a single CPU configuration object.  Multi-CPU platforms
    51     *  would declare multiple Cpu configuration parameters (one per
    52     *  platform CPU).
    53     */
    54    metaonly module Platform inherits xdc.platform.IPlatform
    55    {
    56        /*!
    57         *  ======== BOARD ========
    58         *  This platform's board attributes
    59         */
    60        readonly config xdc.platform.IPlatform.Board BOARD = {
    61            id:             "0",
    62            boardName:      "evmDM365",
    63            boardFamily:    "evmDM365",
    64            boardRevision:  null
    65        };
    66    
    67        readonly config xdc.platform.IExeContext.Cpu CPU = {
    68            id:             "0",
    69            clockRate:      216.0,
    70            catalogName:    "ti.catalog.arm",
    71            deviceName:     "TMS320DM365",
    72            revision:       "",
    73        };
    74    
    75    instance:
    76    
    77        override readonly config xdc.platform.IPlatform.MemoryMap
    78            externalMemoryMap = [
    79                ["DDR",  {name: "DDR",  base: 0x80000000, len: 0x1000000}],
    80            ];
    81    
    82        /*
    83         *  ======== sectMap ========
    84         *  Define a placement of compiler generated output sections into
    85         *  memory regions defined in the memTab above.
    86         */
    87        override config string codeMemory = "DDR";
    88        override config string dataMemory = "DDR";
    89        override config string stackMemory = "DDR";
    90    };
