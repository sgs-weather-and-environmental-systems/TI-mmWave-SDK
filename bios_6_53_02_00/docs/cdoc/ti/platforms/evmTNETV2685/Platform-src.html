    38    /*!
    39     *  ======== Platform ========
    40     *  Platform support for the evmTNETV2685
    41     *
    42     *  This module implements xdc.platform.IPlatform and defines configuration
    43     *  parameters that correspond to this platform's Cpu's, Board's, etc.
    44     *
    45     *  The configuration parameters are initialized in this package's
    46     *  configuration script (package.cfg) and "bound" to the TCOM object
    47     *  model.  Once they are part of the model, these parameters are
    48     *  queried by a program's configuration script.
    49     *
    50     *  This particular platform has a single Cpu, and therefore, only
    51     *  declares a single CPU configuration object.  Multi-CPU platforms
    52     *  would declare multiple Cpu configuration parameters (one per
    53     *  platform CPU).
    54     */
    55    metaonly module Platform inherits xdc.platform.IPlatform
    56    {
    57        readonly config xdc.platform.IPlatform.Board BOARD = {
    58            id:             "0",
    59            boardName:      "evmTNETV2685",
    60            boardFamily:    "evmTNETV2685",
    61            boardRevision:  null,
    62        };
    63    
    64        readonly config xdc.platform.IExeContext.Cpu CPU = {
    65            id:             "0",
    66            clockRate:      500,
    67            catalogName:    "ti.catalog.c6000",
    68            deviceName:     "TMS320CTNETV2685",
    69            revision:       "",
    70        };
    71    
    72    instance:
    73    
    74        override readonly config xdc.platform.IPlatform.Memory
    75            externalMemoryMap[string] = [
    76                ["DDR2", {name: "DDR2", base: 0xE0000000, len: 0x10000000}],
    77            ];
    78    
    79        override config string codeMemory = "DDR2";
    80    
    81        override config string dataMemory = "DDR2";
    82    
    83        override config string stackMemory = "DDR2";
    84    
    85        /*
    86         *  ======== l1PMode ========
    87         *  Define the amount of L1P RAM used for L1 Program Cache.
    88         *
    89         *  Check the device documentation for valid values.
    90         */
    91        config String l1PMode = "32k";
    92    
    93        /*
    94         *  ======== l1DMode ========
    95         *  Define the amount of L1D RAM used for L1 Data Cache.
    96         *
    97         *  Check the device documentation for valid values.
    98         */
    99        config String l1DMode = "32k";
   100    
   101        /*
   102         *  ======== l2Mode ========
   103         *  Define the amount of L2 RAM used for L2 Cache.
   104         *
   105         *  Check the device documentation for valid values.
   106         */
   107        config String l2Mode = "0k";
   108    };
